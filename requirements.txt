# NeuralFn's default install is intentionally lean.
#
# The native/core SDK and compiled CUDA training entrypoints do not need
# Python package dependencies here. Install optional extras explicitly:
#
#   pip install -e ".[tile-cuda]"   # CUDA Tile build helper
#   pip install -e ".[datasets]"    # HF datasets, tiktoken, NumPy
#   pip install -e ".[graph]"       # graph analysis/runtime helpers
#   pip install -e ".[server]"      # FastAPI/editor/MCP backend
#   pip install -r requirements-full.txt
#
# Torch is intentionally not listed in any NeuralFn requirements file.
